## 顺序栈的入栈出栈操作实现

C语言实现：
栈结构
```
typedef int elemtype;
typedef struct {
	elemtype data[100];
	int top;
}sqstack;
```
栈函数
```
//栈初始化
void init(sqstack* &a){
	a=(sqstack *)malloc(sizeof(sqstack));
	a->top=-1;
}
//入栈操作
bool push(sqstack* &a,elemtype e){
	if(a->top>100){
		printf("栈已满");
		return false;
	}
	a->top++;
	a->data[a->top]=e;
	return true;
}
//出栈操作
bool pop(sqstack* &a){
	if(a->top==-1){
		printf("栈为空");
		return false;
	}
	printf("出栈%d",a->data[a->top]);
	a->top--;
	return true;
}

```
测试函数
```
int main(){
	sqstack *a;
	int e;
	init(a);
	printf("请输入入栈元素:");
	scanf("%d",&e);
	push(a,e);
	pop(a);
	return 0;
}
```
